\begin{titledpoem}
    \stanza{
        Siostra White stanęła przeciw doktrynie Trójcy, \\
        Widząc w niej zagrożenie dla wiary dziedzictwa. \\
        Jej słowa ostrzegały przed zmianą fundamentów, \\
        Które Bóg ustanowił dla swoich wybranych ludów.
    }

    \stanza{
        Osobowość Boga to filar naszej wiary, \\
        Nie można go zastąpić przez spekulacji czary. \\
        Doświadczenia przeszłe są skarbem ludu Bożego, \\
        Nie wolno ich odbierać przez nauki porządku nowego.
    }

    \stanza{
        Fundamentalne Zasady przez modlitwę odkryte, \\
        Przez pięćdziesiąt lat były w sercach wyryte. \\
        Czy pozwolimy, by teoria intelektualna, \\
        Zniszczyła to, co prawda niebiańska nam dała?
    }
\end{titledpoem}

% Filary Wiary

\begin{titledpoem}
    \stanza{
        Gdy Kellogg swoje teorie o Bogu przedstawiał, \\
        White jasno przed tą drogą ostrzegała. \\
        Nie można było łączyć panteizmu z prawdą, \\
        Ani Trójcy doktryny z Bożą władzą.
    }

    \stanza{
        Znaki, które nas czynią tym, kim jesteśmy, \\
        Muszą być zachowane, tego się trzymajmy. \\
        Fundamentalne Zasady na autorytecie stoją, \\
        Przed zwodniczymi teoriami nas osłonią.
    }

    \stanza{
        Czy nowa organizacja powstała wśród nas? \\
        Czy filary wiary zburzyliśmy w ten czas? \\
        Czy osobowość Boga jest wciąż jasno znana, \\
        Czy przez doktrynę Trójcy została zapomniana?
    }
\end{titledpoem}

% Wierność Prawdzie

\begin{titledpoem}
    \stanza{
        Alfa błędu w "Living Temple" się zjawiła, \\
        Omega wkrótce miała nadejść, White ostrzegała. \\
        Nie wchodźcie w spór o osobowość Boga, \\
        Bo to jest niebezpieczna, zwodnicza droga.
    }

    \stanza{
        Niektórzy twierdzili, że White popiera te nauki, \\
        Lecz ona zaprzeczyła, odrzucając te wykładni. \\
        Wyrwane z kontekstu słowa mogą zwodzić, \\
        Lecz prawda musi w pełnym świetle chodzić.
    }

    \stanza{
        Trzymajmy się mocno zasad fundamentalnych, \\
        Odkrytych punkt po punkcie w czasach dawnych. \\
        Bo znaki, które nas określają jako lud wybrany, \\
        Muszą być zachowane, jak Bóg nam nakazał.
    }
\end{titledpoem}
