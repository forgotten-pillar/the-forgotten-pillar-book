% Constructive Criticism

\begin{titledpoem}
    \stanza{
        Bóg osobowy na tronie zasiada, \\
        Tak nasza Zasada Fundamentalna powiada. \\
        Wszędzie obecny przez Ducha swojego, \\
        To fundament wiary ludu Bożego.
    }

    \stanza{
        Lecz przyszły słowa, co zdały się mądre, \\
        Subtelna zmiana, co wiary nie oszczędzi. \\
        "Postać Boga poza ludzkim pojmowaniem", \\
        Tajemnica zbyt wielka, by ją nazwać mianem.
    }
\end{titledpoem}

\begin{titledpoem}
    \stanza{
        "Dyskusje o postaci Boga" - pisał Kellogg, \\
        "Bezużyteczne są, bałwochwalstwa to droga". \\
        Filozofia ta, tak gładko przedstawiona, \\
        Była sidłem, w którym dusza uwięziona.
    }

    \stanza{
        Błąd ubrany w prawdę pięknie się prezentuje, \\
        Gdy Pismo wykręcone zręcznie argumentuje. \\
        Mały krok od Zasad, które wyznawaliśmy, \\
        Wielki skok, przez który wiarę utraciliśmy.
    }
\end{titledpoem}

\begin{titledpoem}
    \stanza{
        Strzeż się umysłu, co mądrym się mieni, \\
        By nie dostrzec zwiedzenia w prawdy odzieni. \\
        Bóg jest osobowy, konkretny i realny, \\
        To prawda, którą "Temple" ukrył w sposób fatalny.
    }

    \stanza{
        Fundamenty wiary stoją niewzruszone, \\
        Choć teorie nowe są wciąż przedstawione. \\
        Bóg Ojciec, osobowa duchowa istota, \\
        To prawda, której broni nasza wiara złota.
    }
\end{titledpoem}
