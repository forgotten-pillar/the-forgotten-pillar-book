\begin{titledpoem}
    
    \stanza{
        Gdy szatan zwiedzenia wprowadza wśród nas, \\
        Do źródeł powrócić nadszedł już czas. \\
        Pionierzy wiary drogę wskazali, \\
        Byśmy w prawdzie Bożej wytrwali.
    }

    \stanza{
        Jeżeli z filarów ktoś jeden choć ruszy, \\
        W odstępstwo popadnie przeciwko swej duszy. \\
        Przedrukujmy pisma z dawnych lat, \\
        Pionierów niech usłyszy nasz obecny świat.
    }

    \stanza{
        O osobowości Boga prawda trwa, \\
        Choć wielu ją zmienić ochotę ma. \\
        Zbierzmy światło, które Bóg nam dał, \\
        By fundament wiary na nim mocno stał.
    }

\end{titledpoem}
