\begin{titledpoem}

    \stanza{
        Filary prawdy, starannie wzniesione, \\
        Przez naszych pionierów wymodlone. \\
        Pewne zasady, Boże zamiary, \\
        Na wieki mocna platforma wiary.
    }

    \stanza{
        Strzeżmy się tego, co sprytnie zmienia \\
        Coś, co nie miało być do zmienienia. \\
        Nasza tożsamość ugruntowana \\
        W nich jak w objawieniu od Pana.
    }

    \stanza{
        Wejdźmy więc pewnie na to podłoże, \\
        Gdzie mądrość i światło jaśnieją Boże. \\
        Brońmy tych prawd, póki w nas siła, \\
        By wieczna światłość się z nich przebiła.
    }
    
\end{titledpoem}
