% Fundamentalne Zasady

\begin{titledpoem}
    \stanza{
        Fundament wiary przez Mistrza zbudowany, \\
        Przetrwa nawałnicę, burzę, wszystkie zmiany. \\
        Zasady prawdy, które Bóg objawił nam, \\
        To filar pewny, co nie ugnie się jak kłam.
    }

    \stanza{
        Jeden Bóg, istota duchowa, osobowa, \\
        Przez Ducha Świętego wszędzie obecna, gotowa. \\
        Chrystus, Syn Ojca Wiecznego, przez którego, \\
        Bóg stworzył wszystko z zamysłu mądrego.
    }

    \stanza{
        Gdy człowiek od zasad tych odchodzi śmiało, \\
        Podkopuje fundament, co trwać miało. \\
        Mądrość pokoleń odrzuca bez wahania, \\
        Ścieżkę objawioną przez Boga zasłania.
    }
\end{titledpoem}

% Trzymać się Prawdy

\begin{titledpoem}
    \stanza{
        Trzymajmy się mocno prawdy objawionej, \\
        Przez pionierów w modlitwie odnalezionej. \\
        Punkt po punkcie, w studiach Słowa Bożego, \\
        Odkryli skarb wiary, coś najcenniejszego.
    }

    \stanza{
        Osobowość Boga nie jest mglistą teorią, \\
        Lecz prawdą, co świeci nad życia historią. \\
        Ojciec jako Bóg, Chrystus jako Syn Jego, \\
        To filary wiary ludu adwentowego.
    }

    \stanza{
        Nie dajmy się zwieść zwodniczym naukom, \\
        Co podważają to, co trwało wiekom. \\
        Fundamentalne zasady są jak skała, \\
        Na której wiara nasza zawsze stała.
    }
\end{titledpoem}

% Wierność Fundamentom

\begin{titledpoem}
    \stanza{
        Platforma wiecznej prawdy nas prowadzi, \\
        Kto z niej zstępuje, ten sobie zawadzi. \\
        Pięćdziesiąt lat budowy na fundamencie, \\
        Nie może pójść na marne w jednym momencie.
    }

    \stanza{
        Bóg wzywa nas, byśmy się mocno trzymali, \\
        Zasad, które przodkowie nam przekazali. \\
        Nie jako wyznanie wiary czy dogmaty, \\
        Lecz jako prawdy, co przetrwają lata.
    }

    \stanza{
        Niech osobowość Boga jasno świeci, \\
        W sercach wierzących, dorosłych i dzieci. \\
        Fundament naszej wiary niech trwa wiecznie, \\
        By lud Boży mógł kroczyć bezpiecznie.
    }
\end{titledpoem}
