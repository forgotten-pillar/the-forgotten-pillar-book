% The proper solution

\begin{titledpoem}
    
    \stanza{
        Heed the words of those long past, \\
        Their sacred truths were built to last. \\
        Return to foundations set with care, \\
        Find God's true nature revealed there.
    }

    \stanza{
        When deceptions subtly creep in, \\
        Moving pillars is a grievous sin. \\
        The pioneers' voices still should speak, \\
        Their ancient wisdom we should seek.
    }

    \stanza{
        For Ellen counseled in her day, \\
        From original light do not stray. \\
        God's personality divine, \\
        In pioneer writings clearly shines.
    }
    
\end{titledpoem}
