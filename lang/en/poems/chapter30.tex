% The Final Call

\begin{titledpoem}
    
    \stanza{
        Stand firm on truth, reject false light, \\
        Hold fast to principles that shine so bright. \\
        Wake up, God's servants, sound the alarm, \\
        Guard sacred doctrines from Satan's harm.
    }

    \stanza{
        The Father and Son, distinct and true, \\
        Their personality we must construe. \\
        Not theories strange that mislead the heart, \\
        But truths eternal that set us apart.
    }

    \stanza{
        If we continue in what we first heard, \\
        Faithful to teachings of His holy Word, \\
        Joined with our Maker in heavenly love, \\
        Blessed with the spirit sent from above.
    }

    \stanza{
        The time grows short for this final call, \\
        Choose now the light lest in darkness you fall. \\
        Soon Christ returns on clouds from the sky, \\
        To take His faithful to glory on high.
    }
\end{titledpoem}
