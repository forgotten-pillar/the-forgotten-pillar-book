% The prophetic call for a renewal of the old pillars

\begin{titledpoem}
    \stanza{
        Stare istine moramo ponovno otkriti, \\
        I srca naša njima ispuniti. \\
        Božji put molitvom očistiti trebamo, \\
        Da Njegov Duh u punini primamo.
    }

    \stanza{
        Svaki vjetar nauka danas puše jako, \\
        Obmana se širi podmuklo i lako. \\
        Ali Bog obećava da će istina stajati, \\
        I da će vjerne svoje sačuvati.
    }

    \stanza{
        Rešetanje dolazi kroz svjedočanstvo pravo, \\
        Koje će razlučiti bolesno i zdravo. \\
        Neki će ustati protiv jasne istine, \\
        Dok će vjerni stajati usred tmine.
    }

    \stanza{
        Izravno svjedočanstvo mora biti izneseno, \\
        Kao što je u prošlosti bilo jasno rečeno. \\
        Proučavajmo međaše koje su pioniri postavili, \\
        Njihovo razumijevanje Boga koje su nam ostavili.
    }

    \stanza{
        Zaboravili smo stup o Božjoj ličnosti, \\
        Sad trebamo vratiti se izvoru mudrosti. \\
        Kako nas je Gospodin vodio u prošlosti, \\
        Tako će nas voditi do vječne radosti.
    }

    \stanza{
        Probudimo se i stavimo se pod Božju disciplinu, \\
        Obucimo Božju opremu u svaku istinu. \\
        Jer Bog nas poziva da čvrsto držimo, \\
        Principe vječne dok Njemu služimo.
    }
\end{titledpoem}
