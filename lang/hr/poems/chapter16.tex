% Dr. Kellogg i panteizam

\begin{titledpoem}

    \stanza{
        U prirodi Bog svoju moć pokazuje, \\
        Kroz svaki list i cvijet On djeluje. \\
        No Kellogg je zašao u zabludu tešku, \\
        Kad je trojstvo prihvatio kao stazu nebesku.
    }

    \stanza{
        No greška se u pogledu krije, \\
        Jer Božju ličnost krivo prikazat ne smije. \\
        Jer Bog nije priroda, niti u njoj osobno prebiva, \\
        Već kao osoba na prijestolju počiva.
    }

    \stanza{
        Panteizam je zamka što zavodi um, \\
        Kad Božju prisutnost shvatiš kao sveopći šum. \\
        Njegova prisutnost jest u svemu što diše, \\
        Ali Njegova ličnost na nebu se piše.
    }
    
\end{titledpoem}
