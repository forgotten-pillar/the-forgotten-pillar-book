% Odgovor na Kellogove trinitarijanske sentimente

\begin{titledpoem}

    \stanza{
        Kellogova kriza, više od panteizma bila, \\
        O Božjoj ličnosti istinu je skrila. \\
        Ellen White nam kroz svoje spise piše, \\
        Da Trojstvo s Istinom, se ne slaže više.
    }

    \stanza{
        "Otac i Sin su jedno u cilju, umu," \\
        "Ali ne u osobi," kaže ona u punom razumu. \\
        Fundamentalni Principi to jasno vele, \\
        Dok trinitarijanske zablude, trojedinog Boga žele.
    }

    \stanza{
        Ellen White nas poziva da budemo jaki, \\
        Da na platformi istine stoji vjernik svaki. \\
        Čvrsto stojmo na Istini koju smo primili, \\
        O Božjoj ličnosti koju smo zagovarali.
    }
    
\end{titledpoem}
