% Je li Bog osoba? - članak Johna N. Loughborougha

\begin{titledpoem}
    \stanza{
        Na nebu Bog na prijestolju sjedi, \\
        Tijelo i oblik On zaista vrijedi. \\
        Nije tek duh bez forme i lica, \\
        Već biće stvarno, ne puka pričica.
    }

    \stanza{    
        Njegova slava presjajna je za nas, \\
        Smrtne oči ne podnose taj sjaj i kras. \\
        Ipak kroz Duha svuda je prisutan, \\
        Kao predstavnik Njegov svemoćan i bitan.
    }

    \stanza{
        U Kristu vidimo Očevu sliku pravu, \\
        Savršen odraz, Njegovu osobu i slavu. \\
        Na Božju sliku stvoreni smo mi, \\
        Ne samo duhom, već tijelom također svi.
    }

    \stanza{
        Od praha zemnog oblikova nas On, \\
        Na svoju sliku, to bijaše Njegov zakon. \\
        Osoba prava s tijelom stvarnim jest, \\
        Ne magla bezoblična, kako neki žele svest.
    }

    \stanza{
        Otac je tjelesan, al' nevidljiv sad, \\
        Čeka da čisti srcem vide Ga jednog dan.
    }
\end{titledpoem}
