% The authority of the Fundamental Principles

\begin{titledpoem}
\stanza{
    Temelji vjere čvrsto stoje sad, \\
    Bog ih je sam postavio nekad. \\
    Platforma istine što vječno traje, \\
    Međaši što nam Gospodin ih daje.
}

\stanza{
    Pioniri su tražili u molitvi put, \\
    Točku po točku, Božji duh je tu. \\
    Sestra White, Smith i James zajedno su stali, \\
    Fundamentalne Principe zapisali.
}

\stanza{
    No nakon smrti Ellen White, sve se mijenja, \\
    Trojstvo ulazi u naša vjerovanja. \\
    Wilcox je tiho promijenio nauk stari, \\
    Bez odobrenja crkve, u novoj stvari.
}

\stanza{
    Vratimo se istini koju Bog nam dade, \\
    Ne ljudskim riječima što sumnju sade. \\
    Autoritet principa neupitan je bio, \\
    Jer sam Gospodin ih je ustanovio.
}
\end{titledpoem}
