% The authority of the Fundamental Principles

\begin{titledpoem}
    \stanza{
        Temelji vjere čvrsto stoje sad, \\
        U početku, Bog ih je uspostavio tad. \\
        Platforma istine što vječno traje, \\
        Međaši vjere, što Gospod nam daje.
    }

    \stanza{
        Pioniri naši s molitvom su tražili, \\
        Točku po točku pažljivo složili. \\
        Smith, James i Ellen zajedno su stali, \\
        Fundamentalne Principe zapisali.
    }

    \stanza{
        Al' nakon smrti Ellen White, gle, \\
        Promjene došle, mijenjalo se sve. \\
        Trojstvo uvedeno gdje ne bješe prije, \\
        Istina stara u sjeni se krije.
    }

    \stanza{
        Vratimo se istini što Bog nam dade, \\
        Ne ljudskim riječima što sumnju sade. \\
        Proučimo povijest s molitvom sad, \\
        Da vidimo gdje skrenusmo u pad.
    }
\end{titledpoem}
