% The Sabbath God vs. Sunday God - J. B. Frisbie

\begin{titledpoem}
    \stanza{
        Dva boga, dvije istine se bore, \\
        Jedan stvaran, drugi plod zablude gore. \\
        Bog subote na prijestolju sjedi gore, \\
        Dok bog nedjelje nema tijelo, tako zbore.
    }

    \stanza{
        Duhom svojim ispunjava svemir cijeli, \\
        Ali nije svugdje kao što neki žele. \\
        Osobno biće s tijelom i dijelovima, \\
        Vidljiv onima s čistim srcima.
    }

    \stanza{
        "Tko vidi mene, Oca je vidio" - \\
        Isus je jasno svima objavio. \\
        Bog ima prebivalište na nebu, \\
        Ne ispunjava prostor u svakom trenu.
    }

    \stanza{
        Bog nedjelje je "svugdje prisutan", \\
        Bez tijela, bez dijelova, trojstvom obuzdan. \\
        Iz poganskih izvora ta doktrina teče, \\
        Gdje filozofi o "čistom Duhu" rekoše.
    }

    \stanza{
        Izaberimo Boga koji počinu sedmi dan, \\
        Osobu stvarnu, ne filozofski san. \\
        Jer dan koji svetkujemo otkriva jasno, \\
        Kojem se Bogu klanjamo iskreno i glasno.
    }
\end{titledpoem}
