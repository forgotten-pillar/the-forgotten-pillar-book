\begin{titledpoem}
    \stanza{
        Adventisti danas na raskrižju stoje, \\
        Između prakse stare i istine svoje. \\
        Hoćemo li slijediti pionira put, \\
        Ili disciplinom gušiti Duha kut?
    }

    \stanza{
        Bog će obnoviti temelje naše vjere, \\
        Stupove istine što vrijeme ne bere. \\
        Prepoznajmo vrijeme Božjeg pohođenja, \\
        Prije nego dođe dan konačnog suđenja.
    }

    \stanza{
        Neka savjest bude slobodna i čista, \\
        Podložna samo Riječi Isusa Krista. \\
        Jer vjeća nisu glas Božji za vjerovanje, \\
        Već samo za djela i misije planiranje.
    }
\end{titledpoem}
