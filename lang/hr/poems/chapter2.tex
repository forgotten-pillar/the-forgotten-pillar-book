% The Fundamental Prinicples

\begin{titledpoem}
    \stanza{
        Najveći Radnik temelj je izgradio, \\
        Kroz oluje i bure čvrsto ga postavio. \\
        Principi istine naša su platforma, \\
        Stupovi vjere, sigurna nam norma.
    }

    \stanza{
        Jedan Bog, osobno duhovno biće, \\
        Po Duhu Svetom svugdje prisutan bit će. \\
        Isus Krist, Sin Vječnoga Oca, \\
        Po kojem sve postoji, naša je moća.
    }

    \stanza{
        Kad čovjek od temeljnih principa odstupi, \\
        Prošlo iskustvo naroda Božjeg poruši. \\
        O ličnosti Boga zablude širi, \\
        I prisutnost Njegovu krivo tumači.
    }

    \stanza{
        Čvrstim zahvatom vjere držimo se, \\
        Fundamentalnih principa ne odričemo se. \\
        Jer što je Bog kroz pionire dao, \\
        Za posljednje dane čvrsto je ostalo.
    }
\end{titledpoem}
