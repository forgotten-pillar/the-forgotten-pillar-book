\begin{titledpoem}
    \stanza{
        Sestra White je jasno stala protiv struje, \\
        Dok Trojstvo kao doktrinu odbacuje. \\
        Njene riječi upozorenja svijetle jako, \\
        Protiv učenja što istinu skriva lako.
    }

    \stanza{
        Stupovi vjere, s molitvom građeni, \\
        Sad su izazovom novim ugroženi. \\
        Što je kroz objavu Božju ustanovljeno, \\
        Sad je doktrinarnom promjenom uzdrmano.
    }

    \stanza{
        Ličnost Boga, nekad jasno znana, \\
        Sad je teorijama lažnim zamagljena. \\
        Iskustva prošla naroda Božjeg u opasnosti, \\
        Kad temelji vjere gube na važnosti.
    }

    \stanza{
        Međaši koji nas čine što jesmo, \\
        Trebaju voditi nas sigurno i čvrsto. \\
        Drži se principa fundamentalnih snažno, \\
        Da u magli zablude ne skrenemo pogrešno.
    }
\end{titledpoem}
