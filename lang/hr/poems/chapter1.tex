% Temelj Naše Vjere

\begin{titledpoem}
    \stanza{
        Stupovi istine, s molitvom građeni, \\
        Od pionira vjerno postavljeni. \\
        Principi čvrsti, Božji nacrt sveti, \\
        Temelj jak što vremenu će odoljeti.
    }

    \stanza{
        Čuvaj se suptilnih promjena što zovu, \\
        Da mijenjaju istinu Kristovu. \\
        Naš identitet u njima počiva, \\
        Božja objava što dušu oživljava.
    }

    \stanza{
        Stoj čvrsto na toj stijeni spasa, \\
        Gdje mudrost i svjetlost Božja sjaji jasna. \\
        Brani te istine svom snagom svojom, \\
        Jer vječno svjetlo sja u njima s ljubavlju Božjom.
    }
\end{titledpoem}
