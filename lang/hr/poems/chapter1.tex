% The Foundation of our Faith

\begin{titledpoem}

    \stanza{
        Pedeset godina na temelju tom, \\
        Gradili smo vjeru s molitvom. \\
        Međaši naši ne smiju past', \\
        Jer u njima je Božja vlast.
    }

    \stanza{
        Knjige nove donose mrak, \\
        A istina svijetli kao znak. \\
        O Božjoj ličnosti ne nagađaj, \\
        Već Njegovu riječ proučavaj.
    }

    \stanza{
        Stoj čvrsto na toj stijeni spasa, \\
        Gdje mudrost i svjetlost Božja sjaji jasna. \\
        Iskustvo prošlo ne zaboravi, \\
        Jer Bog nas njime pripravi.
    }
    
\end{titledpoem}
