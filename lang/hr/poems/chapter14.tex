% Adventist pioneers and the Trinity doctrine

\begin{titledpoem}
    \stanza{
        Pioniri čvrsto stajahu protiv Trojstva moći, \\
        Svjedočanstvo jasno kao svjetlost u noći. \\
        Vidjeli su kako doktrina lukavo skriva, \\
        Što Pismo otkriva umu koji istinu otkriva.
    }

    \stanza{
        James White ju nazva među "popularne bajke" znane, \\
        Učenje koje Božju osobnost čini neprepoznane. \\
        Jer Otac i Sin kao odvojena bića stoje, \\
        Ne stopljeni u jedno kako Trojstvo to broje.
    }

    \stanza{
        Dvije zasebne osobe s namjerom ujedinjene, \\
        U duhu i djelu, u umu sjedinjene. \\
        Baš kao učenici u Kristu postaju jedno, \\
        Ali ostaju pojedinci pod Božjim Sinom vrijedno.
    }

    \stanza{
        Duh Sveti, Božja prisutnost božanska, \\
        Nije odvojeno biće trojstvene zamisli sjajna. \\
        Već istinski Božji Duh poslan odozgo, \\
        Očev predstavnik, ljubavi posrednik mnogo.
    }

    \stanza{
        Pismo ne podržava što tradicija je rekla, \\
        Kad u Niceji ova doktrina je potekla. \\
        Ivan sedamnaest razbija trinitarni pogled, \\
        Otkrivajući Oca i Sina kao bića, jasan slijed.
    }

    \stanza{
        Kristovo božanstvo ne počiva u tajanstvenoj trojici, \\
        Već u Njegovom rođenom Sinovstvu vidimo istinu licu. \\
        Savršena slika Očevog lica sjajnog, \\
        Nasljednik punine božanskog blagoslova trajnog.
    }
\end{titledpoem}

\begin{titledpoem}
    \stanza{
        Pioniri su znali što Biblija jasno kaže, \\
        Da je Bog Otac, Biće najdraže. \\
        Osobna, duhovna prisutnost na visini, \\
        Sveprisutan kroz Duha, ipak prebiva u visini.
    }

    \stanza{
        Kelloggova kriza pokazala je istinu pravu, \\
        Ellen White ostala je čvrsta u svom stavu. \\
        Nije prihvatila trojstvenu doktrinu nikada, \\
        Iako mnogi tvrde suprotno i sada.
    }

    \stanza{
        Jedinstvo Oca i Sina nije u osobi jednoj, \\
        Već u svrsi, umu i karakteru vrijednoj. \\
        Kao što Krist i učenici jedno jesu, \\
        A ipak zasebne osobe u svom bijesu.
    }

    \stanza{
        Tako stoji istina protiv zablude duge noći, \\
        Sačuvana od pionira i Sestre White moći. \\
        Ne kroz zbrku ili mistične misli, \\
        Već kroz jasno Pismo istina se čisti.
    }

    \stanza{
        Adventisti danas trebaju razumjeti pravo, \\
        Što su pioniri vjerovali zdravo. \\
        Njihovo razumijevanje Božje osobnosti je čvrsto, \\
        Podržano Pismom, jasno i jednostavno isto.
    }
\end{titledpoem}
