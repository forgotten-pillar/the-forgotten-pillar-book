% Sjećanje na Njegovo učenje u našoj prošlosti

\begin{titledpoem}
    \stanza{
        Temelji vjere naše čvrsto stoje, \\
        Na njima pioniri gradili su svoje. \\
        Fundamentalni principi, Božjom rukom dani, \\
        Stupovi istine, za vječnost sazdani.
    }

    \stanza{
        Pitanje "Tko je Bog?" od ključne je važnosti, \\
        Odgovor traži vjeru, ne ljudske mudrosti. \\
        Je li On Otac, osobno i duhovno biće, \\
        Ili trojstvo, kako novo učenje niče?
    }

    \stanza{
        Spekulacije o Bogu vode u zabludu, \\
        Sjeme nevjere niče u ljudskom razumu. \\
        Gospodin nam je dao jasnu poruku za sada, \\
        Proučavajmo Kristovo učenje, tu je nada.
    }

    \stanza{
        Upozorenja sestre White i danas jasno zvone, \\
        Čuvajte se novih teorija što istinu gone. \\
        Na pijesku nagađanja ne gradite svoj dom, \\
        Već na stijeni istine, objavljenoj Riječju tom.
    }
\end{titledpoem}
