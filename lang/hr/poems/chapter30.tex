% Posljednji poziv

\begin{titledpoem}

    \stanza{
        Mnogi su napustili prvu ljubav svoju, \\
        I izgubili svjetlost u duhovnom boju. \\
        Tamu su svjetlošću nazivali dugo, \\
        A istinu Božju zamijenili za drugo.
    }

    \stanza{
        Fundamentalni principi su naša zaštita, \\
        Bez njih je vjera prazna i razbita. \\
        Ličnost Boga i Sina moramo priznati, \\
        I u njihovoj istini čvrsto ostati.
    }
    
    \stanza{
        Na pravom temelju čvrsto stojimo, \\
        Istine Božje Riječi se ne bojimo. \\
        Lažne teorije odbacujemo smjelo, \\
        Jer Božja istina je naše djelo.
    }

    \stanza{
        Ličnost Boga i Sina ne smijemo poricati, \\
        Već istinu o njima vjerno objavljivati. \\
        Ako u srcu čuvamo što smo prvo čuli, \\
        S Ocem i Sinom bit ćemo ljubavlju povezani.
    }

    \stanza{
        Probudite se, oglašavajte upozorenje, \\
        Jer dolazi vrijeme suda i spasenje. \\
        Kad Krist se vrati na nebeskim oblacima, \\
        Bit ćemo s Njim u vječnim radostima.
    }
    
\end{titledpoem}
