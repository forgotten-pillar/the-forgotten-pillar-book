% The proper solution

\begin{titledpoem}
    
    \stanza{
        Istina davna još uvijek živi, \\
        Glas pionira i danas se divi. \\
        Temelj vjere što Bog nam je dao, \\
        Kroz godine mnoge čvrstim je ostao.
    }

    \stanza{
        Sotona lukavo zablude širi, \\
        Dok istinu staru polako on miri. \\
        Božja ličnost, a ne priroda sama, \\
        U spisima starim otkriva se nama.
    }

    \stanza{
        Ne pomičimo stupove vjere, \\
        Što Duh Sveti postavi bez mjere. \\
        Neka pioniri kroz spise govore, \\
        Svijetlom istine tmine razore.
    }

\end{titledpoem}
