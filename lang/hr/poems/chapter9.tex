% Odgovarajuće Riješenje

\begin{titledpoem}
    \stanza{
        Istina davna što pioniri su znali, \\
        Temelj vjere što su nam predali. \\
        Božja ličnost jasno otkrivena, \\
        U spisima starih svjedoka sačuvana.
    }

    \stanza{
        Sotona lukavo zablude donosi, \\
        Stare stupove vjere ruši i odnosi. \\
        No glas mrtvih još uvijek zbori, \\
        Kroz tiskane riječi istina govori.
    }

    \stanza{
        Vratimo se na put koji Bog je dao, \\
        Kad je narod svoj korak po korak vodio. \\
        Jer istina vremena test će izdržati, \\
        Dok nove teorije neće opstati.
    }
\end{titledpoem}
